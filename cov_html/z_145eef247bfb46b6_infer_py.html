42@task_wrapper 

43def infer(model,image_tensor): 

44 class_labels= ['Beagle', 'Boxer', 'Bulldog', 'Dachshund', 'German_Shepherd', 'Golden_Retriever','Labrador_Retriever', 'Poodle','Rottweiler','Yorkshire_Terrier'] 

45 model.eval() 

46 with torch.no_grad(): 

47 logits = model(image_tensor) 

48 probabilities = F.softmax(logits,dim=-1) 

49 y_pred = probabilities.argmax(dim=-1).item() 

50 

51 confidence = probabilities[0][y_pred].item() 

52 predicted_label = class_labels[y_pred] 

53 return predicted_label,confidence

101def get_latest_checkpoint(base_dir): 

102 base_dir = Path(base_dir) 

103 print(f"Looking for checkpoints starting from directory: {base_dir}") 

104 

105 # Start from the base_dir and search upwards until we find a checkpoint 

106 current_dir = base_dir 

107 while current_dir != current_dir.parent: # Stop when we reach the root directory 

108 checkpoint_pattern = str(current_dir / "**" / "checkpoints" / "*.ckpt") 

109 print(f"Searching with pattern: {checkpoint_pattern}") 

110 

111 checkpoint_files = glob.glob(checkpoint_pattern, recursive=True) 

112 if checkpoint_files: 

113 print(f"Checkpoint files found: {checkpoint_files}") 

114 return max(checkpoint_files, key=os.path.getctime) 

115 

116 current_dir = current_dir.parent 

117 

118 raise FileNotFoundError(f"No checkpoints found in or above {base_dir}")
